```typescript
// Send to OpenAI for analysis

    role: 'system',

    Analyze why certain brands win and what others are missing.

  }, {
    role: 'user',

    ${JSON.stringify(allMentions)}
    
    Identify:

    3. Opportunities (queries where no one dominates)

  }]
});
```

```typescript
// Parse AI analysis
const insights = {
```
